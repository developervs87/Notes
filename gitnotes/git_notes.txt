##Modify terminal and add git branch

alias dk="docker"
alias dk-c="docker-compose"

parse_git_branch() {
        git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/ (\1)/'
}
PS1="\n\[\033[01;34m\]\w \[\033[00m\]\[\033[01;32m\]\$(parse_git_branch)\[\033[00m\] \n\[\033[01;32m\]\u@\h\[\033[00m\]:\$ "
